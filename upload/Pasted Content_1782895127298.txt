    function drawBoardStatic() {
        const c = bctx;
        // Background
        c.fillStyle = '#d6cfc3';
        c.fillRect(0, 0, 600, 600);

        // Home base areas
        for (const color of COLORS) {
            const a = HOME_AREAS[color];
            c.fillStyle = COLOR_HEX[color];
            c.fillRect(a.c1*CS, a.r1*CS, (a.c2-a.c1+1)*CS, (a.r2-a.r1+1)*CS);
            // Inner white box
            const m = CS * 0.8;
            c.fillStyle = '#fff';
            c.fillRect(a.c1*CS + m, a.r1*CS + m, (a.c2-a.c1+1)*CS - 2*m, (a.r2-a.r1+1)*CS - 2*m);
            // Token circles in home
            const positions = HOME_BASE_POS[color];
            positions.forEach(pos => {
                c.beginPath();
                c.arc(pos.x, pos.y, CS*0.35, 0, Math.PI*2);
                c.fillStyle = COLOR_LIGHT[color];
                c.fill();
                c.strokeStyle = COLOR_HEX[color];
                c.lineWidth = 2;
                c.stroke();
            });
        }

        // Path cells
        MAIN_PATH.forEach(([r,col], idx) => {
            const x = col*CS, y = r*CS;
            // Determine cell color
            let fillColor = '#fffdf7';
            // Color start cells
            for (const color of COLORS) {
                if (idx === PLAYER_START_IDX[color]) fillColor = COLOR_LIGHT[color];
            }
            c.fillStyle = fillColor;
            c.fillRect(x+1, y+1, CS-2, CS-2);
            c.strokeStyle = '#d6cfc3';
            c.lineWidth = 0.5;
            c.strokeRect(x+1, y+1, CS-2, CS-2);

            // Star on safe spots
            if (SAFE_SPOTS.has(idx)) {
                drawStar(c, x + CS/2, y + CS/2, 8, 4, 5, '#c8956c');
            }

            // Arrow on start cells
            for (const color of COLORS) {
                if (idx === PLAYER_START_IDX[color]) {
                    c.fillStyle = COLOR_HEX[color] + '44';
                    c.fillRect(x+1, y+1, CS-2, CS-2);
                    // Small arrow indicator
                    c.fillStyle = COLOR_HEX[color];
                    c.font = '14px sans-serif';
                    c.textAlign = 'center';
                    c.textBaseline = 'middle';
                    c.fillText('▶', x+CS/2, y+CS/2);
                }
            }
        });

        // Home stretch cells
        for (const color of COLORS) {
            HOME_STRETCH[color].forEach(([r,col], i) => {
                const x = col*CS, y = r*CS;
                c.fillStyle = COLOR_LIGHT[color];
                c.fillRect(x+1, y+1, CS-2, CS-2);
                c.strokeStyle = COLOR_HEX[color] + '44';
                c.lineWidth = 0.5;
                c.strokeRect(x+1, y+1, CS-2, CS-2);
                // Small triangle pointing to center
                c.fillStyle = COLOR_HEX[color] + '55';
                c.beginPath();
                c.moveTo(x+CS/2, y+CS*0.3);
                c.lineTo(x+CS*0.7, y+CS*0.7);
                c.lineTo(x+CS*0.3, y+CS*0.7);
                c.closePath();
                c.fill();
            });
        }

        // Center area - 4 colored triangles
        const cx = 7.5*CS, cy = 7.5*CS, s = CS*1.5;
        // Red (bottom-left)
        c.fillStyle = COLOR_HEX.red;
        c.beginPath(); c.moveTo(cx,cy); c.lineTo(cx-s,cy+s); c.lineTo(cx-s,cy-s); c.closePath(); c.fill();
        // Green (top-left)
        c.fillStyle = COLOR_HEX.green;
        c.beginPath(); c.moveTo(cx,cy); c.lineTo(cx-s,cy-s); c.lineTo(cx+s,cy-s); c.closePath(); c.fill();

        c.beginPath(); c.moveTo(cx,cy); c.lineTo(cx+s,cy-s); c.lineTo(cx+s,cy+s); c.closePath(); c.fill();
        // Blue (bottom-right)
        c.fillStyle = COLOR_HEX.blue;
        c.beginPath(); c.moveTo(cx,cy); c.lineTo(cx+s,cy+s); c.lineTo(cx-s,cy+s); c.closePath(); c.fill();

        // Center circle
        c.beginPath();
        c.arc(cx, cy, CS*0.55, 0, Math.PI*2);

        c.lineWidth = 2;

        c.font = 'bold 11px "DM Sans", sans-serif';

        c.fillText('HOME', cx, cy);

    function drawStar(c, cx, cy, outerR, innerR, points, color) {
        c.fillStyle = color;
        c.beginPath();
        for (let i = 0; i < points*2; i++) {
            const r = i%2===0 ? outerR : innerR;
            const a = (Math.PI/2*3) + (i * Math.PI / points);
            const x = cx + Math.cos(a)*r;
            const y = cy + Math.sin(a)*r;
            if (i===0) c.moveTo(x,y); else c.lineTo(x,y);
        }
        c.closePath();
        c.fill();
    }

    function getTokenPixelPos(color, tokenIdx) {
        const pIdx = players.findIndex(p => p.color === color);
        if (pIdx === -1) return {x:0,y:0};
        const pos = players[pIdx].tokens[tokenIdx].pos;
        if (pos === -1) {
            return {...HOME_BASE_POS[color][tokenIdx]};
        } else if (pos >= 0 && pos <= 51) {
            const absIdx = (PLAYER_START_IDX[color] + pos) % 52;
            const [r,c] = MAIN_PATH[absIdx];
            return { x: c*CS + CS/2, y: r*CS + CS/2 };
        } else if (pos >= 52 && pos <= 57) {
            const [r,c] = HOME_STRETCH[color][pos - 52];
            return { x: c*CS + CS/2, y: r*CS + CS/2 };
        } else {
            return { x: 7.5*CS, y: 7.5*CS };
        }
    }

    function drawTokens(highlighted = []) {
        // Collect all token positions for overlap detection
        const allPositions = [];
        players.forEach((p, pi) => {
            p.tokens.forEach((t, ti) => {
                if (t.pos === 58) return; // finished tokens shown in home center
                const pos = getTokenPixelPos(p.color, ti);
                allPositions.push({ pi, ti, ...pos, color: p.color });
            });
        });

        // Group by same pixel position
        const groups = {};
        allPositions.forEach(tp => {
            const key = `${Math.round(tp.x)},${Math.round(tp.y)}`;
            if (!groups[key]) groups[key] = [];
            groups[key].push(tp);
        });

        // Draw tokens with offset for overlaps
        const offsets = [[-6,-6],[6,-6],[-6,6],[6,6]];
        Object.values(groups).forEach(group => {
            group.forEach((tp, idx) => {
                let dx = 0, dy = 0;
                if (group.length > 1) {
                    dx = offsets[idx % 4][0];
                    dy = offsets[idx % 4][1];
                }
                const x = tp.x + dx;
                const y = tp.y + dy;
                const isHighlighted = highlighted.some(h => h.pi === tp.pi && h.ti === tp.ti);

                // Glow for highlighted
                if (isHighlighted) {
                    ctx.beginPath();
                    ctx.arc(x, y, CS*0.4, 0, Math.PI*2);
                    ctx.fillStyle = COLOR_HEX[tp.color] + '44';
                    ctx.fill();
                    // Pulsing ring
                    const pulse = Math.sin(Date.now() / 200) * 0.15 + 0.85;
                    ctx.beginPath();
                    ctx.arc(x, y, CS*0.38 * pulse, 0, Math.PI*2);
                    ctx.strokeStyle = '#fff';
                    ctx.lineWidth = 2;
                    ctx.stroke();
                }

                // Token body
                const grad = ctx.createRadialGradient(x-3, y-3, 2, x, y, CS*0.32);
                grad.addColorStop(0, lightenColor(COLOR_HEX[tp.color], 40));
                grad.addColorStop(1, COLOR_HEX[tp.color]);
                ctx.beginPath();
                ctx.arc(x, y, CS*0.3, 0, Math.PI*2);
                ctx.fillStyle = grad;
                ctx.fill();
                ctx.strokeStyle = 'rgba(0,0,0,0.3)';
                ctx.lineWidth = 1.5;
                ctx.stroke();

                // Shine
                ctx.beginPath();
                ctx.arc(x - 3, y - 4, CS*0.1, 0, Math.PI*2);
                ctx.fillStyle = 'rgba(255,255,255,0.35)';
                ctx.fill();
            });
        });

        // Draw finished token count in center
        players.forEach(p => {
            const finished = p.tokens.filter(t => t.pos === 58).length;
            if (finished > 0) {
                const angle = {red: Math.PI*0.75, green: Math.PI*1.25, yellow: Math.PI*1.75, blue: Math.PI*0.25}[p.color];
                const r = CS * 0.3;
                const cx = 7.5*CS + Math.cos(angle)*r;
                const cy = 7.5*CS + Math.sin(angle)*r;
                ctx.beginPath();
                ctx.arc(cx, cy, 5, 0, Math.PI*2);
                ctx.fillStyle = COLOR_HEX[p.color];
                ctx.fill();
            }
        });
    }

    function lightenColor(hex, percent) {
        const num = parseInt(hex.replace('#',''), 16);
        const r = Math.min(255, (num >> 16) + percent);
        const g = Math.min(255, ((num >> 8) & 0xff) + percent);
        const b = Math.min(255, (num & 0xff) + percent);
        return `rgb(${r},${g},${b})`;
    }

    function drawBoard(highlighted = []) {
        ctx.drawImage(boardBuffer, 0, 0);
        drawTokens(highlighted);
    }

    drawBoardStatic();
    drawBoard();

    /* ================================================================
       GAME LOGIC
    ================================================================ */
    function getValidMoves(pIdx, dice) {
        const p = players[pIdx];
        const moves = [];
        p.tokens.forEach((t, ti) => {
            if (t.pos === 58) return; // already finished
            if (t.pos === -1) {
                // In home - need a 6 to come out
                if (dice === 6) {
                    moves.push({ tokenIdx: ti, newPos: 0, type: 'enter' });
                }
            } else {
                const newPos = t.pos + dice;
                if (newPos <= 58) {
                    // Check if a friendly token is already at the destination on main path
                    if (newPos <= 51) {
                        const absNew = (PLAYER_START_IDX[p.color] + newPos) % 52;
                        let blocked = false;
                        p.tokens.forEach((ot, oi) => {
                            if (oi === ti) return;
                            if (ot.pos >= 0 && ot.pos <= 51) {
                                const absOther = (PLAYER_START_IDX[p.color] + ot.pos) % 52;
                                if (absOther === absNew) blocked = true;
                            }
                        });
                        if (!blocked) moves.push({ tokenIdx: ti, newPos, type: 'move' });
                    } else {
                        moves.push({ tokenIdx: ti, newPos, type: 'move' });
                    }
                }
            }
        });
        return moves;
    }

    function checkCapture(pIdx, tokenIdx, doCapture = true) {
        const p = players[pIdx];
        const t = p.tokens[tokenIdx];
        if (t.pos < 0 || t.pos > 51) return null;
        const absPos = (PLAYER_START_IDX[p.color] + t.pos) % 52;
        if (SAFE_SPOTS.has(absPos)) return null;

        let captured = null;
        players.forEach((op, oi) => {
            if (oi === pIdx) return;
            op.tokens.forEach((ot, oti) => {
                if (ot.pos < 0 || ot.pos > 51) return;
                const otherAbs = (PLAYER_START_IDX[op.color] + ot.pos) % 52;
                if (otherAbs === absPos && doCapture) {
                    ot.pos = -1;
                    captured = { playerIdx: oi, tokenIdx: oti, color: op.color };
                    playSound('capture');
                    showToast(`${p.name} captured ${op.name}'s token!`);
                }
            });
        });
        return captured;
    }

    function checkWin(pIdx) {
        return players[pIdx].tokens.every(t => t.pos === 58);
    }

    function executeMove(pIdx, move, callback) {
        isAnimating = true;
        const p = players[pIdx];
        const t = p.tokens[move.tokenIdx];
        const fromPos = t.pos;
        const toPos = move.newPos;

        if (move.type === 'enter') {
            // Animate from home to start
            t.pos = 0;
            playSound('move');
            drawBoard();
            updatePlayerCards();
            setTimeout(() => {
                checkCapture(pIdx, move.tokenIdx);
                drawBoard();
                updatePlayerCards();
                isAnimating = false;
                callback();
            }, 200);
        } else {
            // Step-by-step animation
            let step = 0;
            const totalSteps = toPos - fromPos;
            function doStep() {
                step++;
                t.pos = fromPos + step;
                if (step < totalSteps) {
                    playSound('move');
                    drawBoard();
                    updatePlayerCards();
                    setTimeout(doStep, 120);
                } else {
                    // Final step
                    if (t.pos === 58) {
                        playSound('home');
                        showToast(`${p.name} got a token home!`);
                    } else {
                        playSound('move');
                    }
                    checkCapture(pIdx, move.tokenIdx);
                    drawBoard();
                    updatePlayerCards();
                    isAnimating = false;
                    callback();
                }
            }
            if (totalSteps > 0) doStep();
            else { isAnimating = false; callback(); }
        }
    }

    function nextTurn() {
        consecutiveSixes = 0;
        currentPlayer = (currentPlayer + 1) % activePlayerOrder.length;
        startTurn();
    }

    function startTurn() {
        if (gamePhase === 'finished') return;
        updateTurnIndicator();
        updatePlayerCards();
        drawBoard();
        document.getElementById('dice-msg').textContent = '';

        const pIdx = activePlayerOrder[currentPlayer];
        const p = players[pIdx];

        if (p.isAI) {
            gamePhase = 'idle';
            document.getElementById('roll-btn').disabled = true;
            setTimeout(() => aiTurn(), 700);
        } else {
            gamePhase = 'waiting_roll';
            document.getElementById('roll-btn').disabled = false;
            document.getElementById('dice-msg').textContent = 'Your turn — roll the dice';
        }
    }

    function humanRoll() {
        if (gamePhase !== 'waiting_roll') return;
        rollDiceAnim((value) => {
            handlePostRoll(activePlayerOrder[currentPlayer], value);
        });
    }

    function handlePostRoll(pIdx, value) {
        const p = players[pIdx];
        if (value === 6) {
            consecutiveSixes++;
            if (consecutiveSixes >= 3) {
                document.getElementById('dice-msg').textContent = 'Three 6s! Turn lost.';
                showToast(`${p.name} rolled three 6s — turn forfeited!`);
                setTimeout(nextTurn, 1200);
                return;
            }
        } else {
            consecutiveSixes = 0;
        }

        validMoves = getValidMoves(pIdx, value);
        if (validMoves.length === 0) {
            document.getElementById('dice-msg').textContent = 'No valid moves';
            setTimeout(nextTurn, 1000);
            return;
        }

        if (p.isAI) {
            const move = aiChooseMove(pIdx, value, validMoves);
            executeMove(pIdx, move, () => {
                if (checkWin(pIdx)) {
                    gamePhase = 'finished';
                    showWin(pIdx);
                    return;
                }
                if (value === 6 && consecutiveSixes < 3) {
                    setTimeout(() => aiTurn(), 600);
                } else {
                    setTimeout(nextTurn, 400);
                }
            });
        } else {
            if (validMoves.length === 1) {
                // Auto-move if only one option
                gamePhase = 'animating';
                document.getElementById('dice-msg').textContent = 'Moving...';
                executeMove(pIdx, validMoves[0], () => {
                    if (checkWin(pIdx)) { gamePhase = 'finished'; showWin(pIdx); return; }
                    if (value === 6 && consecutiveSixes < 3) {
                        gamePhase = 'waiting_roll';
                        document.getElementById('roll-btn').disabled = false;
                        document.getElementById('dice-msg').textContent = 'Rolled 6 — roll again!';
                    } else {
                        nextTurn();
                    }
                });
            } else {
                gamePhase = 'waiting_token';
                document.getElementById('dice-msg').textContent = 'Select a token to move';
                canvas.classList.add('clickable');
                // Highlight valid tokens
                highlightLoop();
            }
        }
    }

    let highlightRAF = null;
    function highlightLoop() {
        if (gamePhase !== 'waiting_token') { canvas.classList.remove('clickable'); return; }
        const highlighted = validMoves.map(m => ({ pi: activePlayerOrder[currentPlayer], ti: m.tokenIdx }));
        drawBoard(highlighted);
        highlightRAF = requestAnimationFrame(highlightLoop);
    }

    // Canvas click handler for human token selection
    canvas.addEventListener('click', (e) => {
        if (gamePhase !== 'waiting_token') return;
        const rect = canvas.getBoundingClientRect();
        const scaleX = 600 / rect.width;
        const scaleY = 600 / rect.height;
        const mx = (e.clientX - rect.left) * scaleX;
        const my = (e.clientY - rect.top) * scaleY;

        const pIdx = activePlayerOrder[currentPlayer];
        let clickedMove = null;
        let minDist = Infinity;

        validMoves.forEach(move => {
            const pos = getTokenPixelPos(players[pIdx].color, move.tokenIdx);
            const dx = mx - pos.x, dy = my - pos.y;
            const dist = Math.sqrt(dx*dx + dy*dy);
            if (dist < CS * 0.5 && dist < minDist) {
                minDist = dist;
                clickedMove = move;
            }
        });

        if (clickedMove) {
            if (highlightRAF) cancelAnimationFrame(highlightRAF);
            canvas.classList.remove('clickable');
            gamePhase = 'animating';
            const value = diceValue;
            executeMove(pIdx, clickedMove, () => {
                if (checkWin(pIdx)) { gamePhase = 'finished'; showWin(pIdx); return; }
                if (value === 6 && consecutiveSixes < 3) {
                    gamePhase = 'waiting_roll';
                    document.getElementById('roll-btn').disabled = false;
                    document.getElementById('dice-msg').textContent = 'Rolled 6 — roll again!';
                } else {
                    nextTurn();
                }
            });
        }
    });

    /* ================================================================
       AI LOGIC
    ================================================================ */
    function aiTurn() {
        if (gamePhase === 'finished') return;
        const pIdx = activePlayerOrder[currentPlayer];
        rollDiceAnim((value) => {
            handlePostRoll(pIdx, value);
        });
    }

    function aiChooseMove(pIdx, dice, moves) {
        const p = players[pIdx];
        // Priority 1: Capture if possible
        for (const move of moves) {
            if (move.type === 'move' && move.newPos <= 51) {
                const absPos = (PLAYER_START_IDX[p.color] + move.newPos) % 52;
                if (SAFE_SPOTS.has(absPos)) continue;
                let canCapture = false;
                players.forEach((op, oi) => {
                    if (oi === pIdx) return;
                    op.tokens.forEach(ot => {
                        if (ot.pos >= 0 && ot.pos <= 51) {
                            const otherAbs = (PLAYER_START_IDX[op.color] + ot.pos) % 52;
                            if (otherAbs === absPos) canCapture = true;
                        }
                    });
                });
                if (canCapture) return move;
            }
        }
        // Priority 2: Reach home
        for (const move of moves) {
            if (move.newPos === 58) return move;
        }
        // Priority 3: Enter home stretch if close
        for (const move of moves) {
            if (move.newPos >= 52 && move.newPos < 58) return move;
        }
        // Priority 4: Bring out token on 6
        if (dice === 6) {
            const enterMove = moves.find(m => m.type === 'enter');
            if (enterMove && p.tokens.filter(t => t.pos === -1).length > 2) return enterMove;
        }
        // Priority 5: Move the furthest token
        let bestMove = moves[0];
        let bestPos = -999;
        moves.forEach(move => {
            if (move.newPos > bestPos) {
                bestPos = move.newPos;
                bestMove = move;
            }
        });
        // Random factor to avoid predictability
        if (moves.length > 1 && Math.random() < 0.2) {
            return moves[Math.floor(Math.random() * moves.length)];
        }
        return bestMove;
    }

    function playSound(type) {
        try {
            const ac = getAudioCtx();
            if (ac.state === 'suspended') ac.resume();
            const osc = ac.createOscillator();
            const gain = ac.createGain();
            osc.connect(gain);
            gain.connect(ac.destination);

            switch(type) {
                case 'dice':
                    osc.type = 'square';
                    osc.frequency.setValueAtTime(200, ac.currentTime);
                    osc.frequency.exponentialRampToValueAtTime(80, ac.currentTime + 0.1);
                    gain.gain.setValueAtTime(0.08, ac.currentTime);
                    gain.gain.exponentialRampToValueAtTime(0.001, ac.currentTime + 0.12);
                    osc.start(ac.currentTime);
                    osc.stop(ac.currentTime + 0.12);
                    break;
                case 'move':
                    osc.type = 'sine';
                    osc.frequency.setValueAtTime(600, ac.currentTime);
                    gain.gain.setValueAtTime(0.04, ac.currentTime);
                    gain.gain.exponentialRampToValueAtTime(0.001, ac.currentTime + 0.06);
                    osc.start(ac.currentTime);
                    osc.stop(ac.currentTime + 0.06);
                    break;
                case 'capture':
                    osc.type = 'sawtooth';
                    osc.frequency.setValueAtTime(400, ac.currentTime);
                    osc.frequency.exponentialRampToValueAtTime(100, ac.currentTime + 0.3);
                    gain.gain.setValueAtTime(0.1, ac.currentTime);
                    gain.gain.exponentialRampToValueAtTime(0.001, ac.currentTime + 0.3);
                    osc.start(ac.currentTime);
                    osc.stop(ac.currentTime + 0.3);
                    break;
                case 'home':
                    osc.type = 'sine';
                    osc.frequency.setValueAtTime(523, ac.currentTime);
                    osc.frequency.setValueAtTime(659, ac.currentTime + 0.1);
                    osc.frequency.setValueAtTime(784, ac.currentTime + 0.2);
                    gain.gain.setValueAtTime(0.08, ac.currentTime);
                    gain.gain.exponentialRampToValueAtTime(0.001, ac.currentTime + 0.35);
                    osc.start(ac.currentTime);
                    osc.stop(ac.currentTime + 0.35);
                    break;
                case 'win':
                    [523,659,784,1047].forEach((freq, i) => {
                        const o = ac.createOscillator();
                        const g = ac.createGain();
                        o.connect(g); g.connect(ac.destination);
                        o.type = 'sine';
                        o.frequency.setValueAtTime(freq, ac.currentTime + i*0.15);
                        g.gain.setValueAtTime(0.08, ac.currentTime + i*0.15);
                        g.gain.exponentialRampToValueAtTime(0.001, ac.currentTime + i*0.15 + 0.4);
                        o.start(ac.currentTime + i*0.15);
                        o.stop(ac.currentTime + i*0.15 + 0.4);
                    });
                    return;
            }
        } catch(e) {}
    }

    let spotifyToken = null;
    let spotifyPlayer = null;
    let spotifyDeviceId = null;
    let currentTrackUri = null;
    let searchResults = [];

    // Check for token in URL hash on page load
    (function checkHashToken() {
        const hash = window.location.hash;
        if (hash && hash.includes('access_token')) {
            const params = new URLSearchParams(hash.substring(1));
            const token = params.get('access_token');
            if (token) {
                spotifyToken = token;
                window.history.replaceState(null, '', window.location.pathname + window.location.search);
                onSpotifyConnected();
            }
        }
    })();

    function initSpotifyAuth() {
        const clientId = document.getElementById('sp-client-id').value || document.getElementById('setup-spotify-id').value;
        if (!clientId) { showToast('Please enter a Spotify Client ID'); return; }
        const redirectUri = encodeURIComponent(window.location.origin + window.location.pathname);
        const scopes = encodeURIComponent('streaming user-read-email user-read-private user-read-playback-state user-modify-playback-state playlist-read-private');
        window.location.href = `https://accounts.spotify.com/authorize?response_type=token&client_id=${clientId}&redirect_uri=${redirectUri}&scope=${scopes}&show_dialog=false`;
    }

    function onSpotifyConnected() {
        document.getElementById('sp-disconnected').classList.add('hidden');
        document.getElementById('sp-connected').classList.remove('hidden');
        document.getElementById('spotify-panel').classList.remove('collapsed');
        tryInitWebPlayback();
        fetchNowPlaying();
    }

    function disconnectSpotify() {
        spotifyToken = null;
        spotifyDeviceId = null;
        if (spotifyPlayer) { spotifyPlayer.disconnect(); spotifyPlayer = null; }
        document.getElementById('sp-disconnected').classList.remove('hidden');
        document.getElementById('sp-connected').classList.add('hidden');
        showToast('Disconnected from Spotify');
    }

    function tryInitWebPlayback() {
        if (typeof Spotify === 'undefined') {
            // Load SDK dynamically
            const s = document.createElement('script');
            s.src = 'https://sdk.scdn.co/spotify-player.js';
            s.onload = () => initWebPlayback();
            s.onerror = () => console.log('Spotify SDK failed to load');
            document.head.appendChild(s);
        } else {
            initWebPlayback();
        }
    }

    function initWebPlayback() {
        if (typeof Spotify === 'undefined' || !Spotify.Player) return;
        spotifyPlayer = new Spotify.Player({
            name: 'LUDO Game',
            getOAuthToken: cb => cb(spotifyToken),
            volume: 0.7
        });
        spotifyPlayer.connect();
        spotifyPlayer.on('ready', ({device_id}) => {
            spotifyDeviceId = device_id;
            console.log('Spotify Web Playback ready, device:', device_id);
        });
        spotifyPlayer.on('player_state_changed', state => {
            if (state) updateNowPlayingUI(state);
        });
        spotifyPlayer.on('authentication_error', () => {
            showToast('Spotify auth expired — please reconnect');
            disconnectSpotify();
        });
    }

    async function fetchNowPlaying() {
        if (!spotifyToken) return;
        try {
            const res = await fetch('https://api.spotify.com/v1/me/player', {
                headers: { 'Authorization': 'Bearer ' + spotifyToken }
            });
            if (res.ok) {
                const data = await res.json();
                updateNowPlayingUI(data);
            }
        } catch(e) {}
    }

    function updateNowPlayingUI(state) {
        if (!state) return;
        const track = state.item;
        if (track) {
            document.getElementById('sp-track-name').textContent = track.name;
            document.getElementById('sp-artist').textContent = track.artists.map(a=>a.name).join(', ');
            const img = document.getElementById('sp-album-art');
            if (track.album && track.album.images && track.album.images[0]) {
                img.src = track.album.images[0].url;
                img.style.display = 'block';
            } else {
                img.style.display = 'none';
            }
            currentTrackUri = track.uri;
        }
        const isPlaying = !state.paused;
        document.getElementById('sp-play-btn').innerHTML = isPlaying
            ? '<i class="fa-solid fa-pause"></i>'
            : '<i class="fa-solid fa-play"></i>';
    }

    async function searchSpotify() {
        const query = document.getElementById('sp-search').value.trim();
        if (!query || !spotifyToken) return;
        const container = document.getElementById('sp-results');
        container.innerHTML = '<div class="text-xs p-2" style="color:#666;">Searching...</div>';
        try {
            const res = await fetch(`https://api.spotify.com/v1/search?q=${encodeURIComponent(query)}&type=track&limit=8`, {
                headers: { 'Authorization': 'Bearer ' + spotifyToken }
            });
            const data = await res.json();
            searchResults = data.tracks ? data.tracks.items : [];
            container.innerHTML = searchResults.length === 0
                ? '<div class="text-xs p-2" style="color:#666;">No results found</div>'
                : searchResults.map((t, i) => `
                    <div class="sp-track" onclick="playSpotifyTrack(${i})">
                        <img src="${t.album.images[0]?.url || ''}" alt="" onerror="this.style.display='none'">
                        <div class="flex-1 min-w-0">
                            <div class="text-sm truncate" style="color:#fff;">${t.name}</div>
                            <div class="text-xs truncate" style="color:#b3b3b3;">${t.artists.map(a=>a.name).join(', ')}</div>
                        </div>
                        <span class="text-xs" style="color:#666;">${formatDuration(t.duration_ms)}</span>
                    </div>
                `).join('');
        } catch(e) {
            container.innerHTML = '<div class="text-xs p-2" style="color:#e63946;">Search failed</div>';
        }
    }

    function formatDuration(ms) {
        const s = Math.floor(ms/1000);
        return Math.floor(s/60) + ':' + String(s%60).padStart(2,'0');
    }

    async function playSpotifyTrack(idx) {
        const track = searchResults[idx];
        if (!track || !spotifyToken) return;
        currentTrackUri = track.uri;
        const body = spotifyDeviceId
            ? { uris: [track.uri], device_id: spotifyDeviceId }
            : { uris: [track.uri] };
        try {
            let url = 'https://api.spotify.com/v1/me/player/play';
            if (spotifyDeviceId) url += `?device_id=${spotifyDeviceId}`;
            const res = await fetch(url, {
                method: 'PUT',
                headers: {
                    'Authorization': 'Bearer ' + spotifyToken,
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({ uris: [track.uri] })
            });
            if (!res.ok && res.status !== 204) {
                // Try without device_id — play on active device
                const res2 = await fetch('https://api.spotify.com/v1/me/player/play', {
                    method: 'PUT',
                    headers: {
                        'Authorization': 'Bearer ' + spotifyToken,
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({ uris: [track.uri] })
                });
                if (!res2.ok && res2.status !== 204) {
                    showToast('Could not play — make sure Spotify is open on a device');
                }
            }
            setTimeout(fetchNowPlaying, 500);
        } catch(e) {
            showToast('Playback failed');
        }
    }
